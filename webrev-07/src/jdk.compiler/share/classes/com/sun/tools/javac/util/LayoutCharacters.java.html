26 package com.sun.tools.javac.util;
27 
28 /** An interface containing layout character constants used in Java
29  *  programs.
30  *
31  *  <p><b>This is NOT part of any supported API.
32  *  If you write code that depends on this, you do so at your own risk.
33  *  This code and its internal interfaces are subject to change or
34  *  deletion without notice.</b>
35  */
36 public interface LayoutCharacters {
37 
38     /** Tabulator column increment.
39      */
40     final static int TabInc = 8;
41 
42     /** Standard indentation for subdiagnostics
43      */
44     final static int DiagInc = 4;
45 
46     /** Standard indentation for additional diagnostic lines
47      */
48     final static int DetailsInc = 2;
49 
50     /** Tabulator character.
51      */
52     final static byte TAB   = 0x9;
53 
54     /** Line feed character.
55      */
56     final static byte LF    = 0xA;
57 
58     /** Form feed character.
59      */
60     final static byte FF    = 0xC;
61 
62     /** Carriage return character.
63      */
64     final static byte CR    = 0xD;
65 
66     /** End of input character.  Used as a sentinel to denote the
67      *  character one beyond the last defined character in a
68      *  source file.
69      */
70     final static byte EOI   = 0x1A;
71 
72     /** Bump column to the next tab.
73      */
74     static int tabulate(int column) {
75         return (column / TabInc * TabInc) + TabInc;
76     }
77 }
